const apiKey = 'key'; document.getElementById('cityInput').addEventListener('keydown', function(event) { if (event.key === 'Enter') { searchPlaces(); } }); document.querySelector('input').addEventListener('dblclick', () => { document.querySelector('input').value = ''; }); document.querySelector('input').addEventListener('mouseover', () => { document.querySelector('input').focus(); }); async function searchPlaces() { const city = document.getElementById('cityInput').value; const resultContainer = document.getElementById('resultContainer'); const loadingMessage = document.getElementById('loading'); resultContainer.innerHTML = ''; // مسح النتائج القديمة loadingMessage.style.display = 'block'; // عرض رسالة التحميل if (!city) { alert("الرجاء إدخال اسم المدينة"); loadingMessage.style.display = 'none'; // إخفاء النص بعد انتهاء التحميل في حالة عدم وجود المدينة return; } try { const geoRes = await fetch(`https://api.geoapify.com/v1/geocode/search?text=${city}&apiKey=${apiKey}`); const geoData = await geoRes.json(); if (!geoData.features.length) { alert("لم يتم العثور على المدينة."); loadingMessage.style.display = 'none'; // إخفاء رسالة التحميل return; } const lat = geoData.features[0].geometry.coordinates[1]; const lon = geoData.features[0].geometry.coordinates[0]; const categories = [ { key: 'catering.restaurant', title: '🍽️ المطاعم', class: 'restaurant'}, { key: 'entertainment', title: '🎉 الترفيه' , class: 'entertainment' }, { key: 'building.commercial', title: '🏢 المباني التجارية', class: 'building' }, ]; let hasResults = false; // لتتبع إذا كانت هناك أماكن for (const category of categories) { const apiUrl = `https://api.geoapify.com/v2/places?categories=${category.key}&filter=circle:${lon},${lat},7000&limit=5&apiKey=${apiKey}`; const res = await fetch(apiUrl); const data = await res.json(); const places = data.features; if (places.length > 0) { const title = document.createElement('h3'); title.className = 'category-title'; title.textContent = category.title; resultContainer.appendChild(title); places.forEach(place => { const name = place.properties.name || 'اسم غير متوفر'; const address = place.properties.formatted || 'عنوان غير متوفر'; const category = place.properties.categories[1] || 'فئة غير متوفرة'; // عرض الفئة الفرعية const openingHours = place.properties.opening_hours || 'غير متوفرة'; const cuisine = (place.properties.catering && place.properties.catering.cuisine) || 'غير محدد'; const card = document.createElement('div'); const categoryClass = category.toLowerCase().replace(/\s+/g, '-'); // لتحويل الفئة إلى lowercase وتغيير الفراغات إلى شرطات const cuisineList = cuisine.split(';'); const type = category === 'building.commercial' ? `النوع: ${place.properties.type || 'غير محدد'}
` : ''; const formattedCuisine = cuisineList.map(item => `
  • ${item}
    • `).join(''); card.className = `card ${categoryClass}` card.innerHTML = `
    الاسم: ${name}
    العنوان: ${address}
    الفئة: ${category}
    ${category === 'catering.restaurant' ? ` أوقات العمل: ${openingHours}
    نوع المأكولات:
      ${formattedCuisine}
    ` : ''} ${category.key === 'building.commercial' ? `أوقات العمل: ${openingHours}
    ` : ''} ${category === 'building.commercial' ? ` أوقات العمل: ${openingHours}
    ` : ''} ${type} عرض على الخريطة
    `; resultContainer.appendChild(card); }); hasResults = true; // تم العثور على أماكن في هذه الفئة } } if (!hasResults) { // إذا لم يتم العثور على أي أماكن في أي فئة، أظهر رسالة للمستخدم const noResultsMessage = document.createElement('p'); noResultsMessage.textContent = "لا توجد أماكن في هذه الفئات."; resultContainer.appendChild(noResultsMessage); } } catch (error) { console.error(error); alert("حدث خطأ أثناء تحميل البيانات."); } finally { loadingMessage.style.display = 'none'; // إخفاء رسالة التحميل بعد الانتهاء } }
